# 游戏管理模块实现方案

## 模块目标

作为系统的"大脑"，负责：
1. 管理游戏状态机
2. 协调各模块之间的调用
3. 处理用户交互（触摸事件）
4. 控制游戏流程（开始→设置目标→游戏→完成→重新开始）

## 各状态实现

### STATE_INIT

```c
static void state_init(void)
{
    ESP_LOGI(TAG, "=== STATE: INIT ===");

    // 1. 初始化渲染器
    renderer_config_t render_cfg = {
        .screen_w = BSP_LCD_H_RES,   // 720
        .screen_h = BSP_LCD_V_RES,   // 1280
        .map_display_size = MAP_DISPLAY_SIZE,  // 640
        .map_display_x = MAP_DISPLAY_X,        // (720-640)/2 = 40
        .map_display_y = MAP_DISPLAY_Y,        // 上方适当偏移
    };
    renderer_init(&render_cfg);

    // 2. 初始化检测模块
    detection_init();

    // 3. 初始化物理引擎（暂不设置物体）
    physics_config_t phys_cfg = {
        .pixels_per_meter = PIXELS_PER_METER,
        .global_friction = GLOBAL_FRICTION,
        .gravity_magnitude = GRAVITY_MAGNITUDE,
        .map_width = MAP_SIZE,
        .map_height = MAP_SIZE,
        .update_rate_hz = PHYSICS_HZ,
    };
    physics_init(&phys_cfg);

    // 4. 初始化相机
    esp_err_t ret = camera_init(NULL);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "Camera init failed: %s", esp_err_to_name(ret));
        renderer_update_status("Camera init FAILED!");
        return;
    }

    // 5. 显示"准备中"画面
    renderer_update_status("Initializing...");

    // 进入下一状态
    change_state(GAME_STATE_CAPTURE);
}
```

### STATE_CAPTURE

```c
static void state_capture(void)
{
    ESP_LOGI(TAG, "=== STATE: CAPTURE ===");

    renderer_update_status("Capturing...");

    // 给相机一点时间稳定（首次启动需要）
    vTaskDelay(pdMS_TO_TICKS(500));

    // 捕获一帧
    camera_frame_t frame;
    esp_err_t ret = camera_capture_frame(&frame, 3000);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "Capture failed: %s", esp_err_to_name(ret));
        renderer_update_status("Capture FAILED! Tap to retry");
        // 等待用户点击重试
        return;
    }

    ESP_LOGI(TAG, "Frame captured: %dx%d, %u bytes",
             frame.width, frame.height, (unsigned)frame.buf_len);

    // 保存帧数据到全局上下文
    g_ctx.captured_frame = frame;

    change_state(GAME_STATE_DETECT);
}
```

### STATE_DETECT

```c
static void state_detect(void)
{
    ESP_LOGI(TAG, "=== STATE: DETECT ===");

    renderer_update_status("Detecting objects...");

    // 运行检测
    detection_result_t det_result;
    esp_err_t ret = detection_run(
        g_ctx.captured_frame.buffer,
        g_ctx.captured_frame.width,
        g_ctx.captured_frame.height,
        &det_result
    );

    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "Detection failed");
        renderer_update_status("Detection FAILED! Tap to retry");
        return;
    }

    ESP_LOGI(TAG, "Detected %u objects", det_result.count);
    for (int i = 0; i < det_result.count; i++) {
        ESP_LOGI(TAG, "  [%d] %s at (%u,%u) size %ux%u, bounce=%.2f friction=%.2f",
                 i, det_result.objects[i].name,
                 det_result.objects[i].x, det_result.objects[i].y,
                 det_result.objects[i].w, det_result.objects[i].h,
                 det_result.objects[i].physics.bounce_coeff,
                 det_result.objects[i].physics.friction_coeff);
    }

    // 保存检测结果
    g_ctx.det_result = det_result;

    change_state(GAME_STATE_MAP_SETUP);
}
```

### STATE_MAP_SETUP

```c
static void state_map_setup(void)
{
    ESP_LOGI(TAG, "=== STATE: MAP_SETUP ===");

    renderer_update_status("Generating map...");

    // 生成游戏地图
    esp_err_t ret = game_map_generate(
        g_ctx.captured_frame.buffer,
        g_ctx.captured_frame.width,
        g_ctx.captured_frame.height,
        &g_ctx.det_result,
        &g_ctx.game_map
    );

    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "Map generation failed");
        renderer_update_status("Map generation FAILED!");
        return;
    }

    // 设置渲染背景
    renderer_set_map_background(&g_ctx.game_map);

    // 将地图物体传递给物理引擎
    physics_set_objects(g_ctx.game_map.objects, g_ctx.game_map.object_count);

    // 设置弹珠初始位置（地图中心偏上）
    float marble_init_x = MAP_SIZE / 2.0f;
    float marble_init_y = MAP_SIZE * 0.2f;
    physics_set_marble_initial(marble_init_x, marble_init_y, MARBLE_RADIUS);

    // 初始渲染弹珠
    const marble_physics_t *m = physics_get_marble();
    marble_render_info_t info = {
        .map_x = m->pos_x, .map_y = m->pos_y, .radius = m->radius
    };
    renderer_update_marble(&info);

    renderer_update_status("Tap screen to set target");
    change_state(GAME_STATE_SET_TARGET);
}
```

### STATE_SET_TARGET

```c
static void state_set_target(void)
{
    // 此状态由触摸事件驱动，不主动执行逻辑
    // 等待用户在屏幕上点击设定目标

    // 显示提示文字
    renderer_update_status("Tap screen to set target destination");

    // 触摸处理在事件回调中：
    // - 将屏幕坐标转为地图坐标
    // - 检查是否在有效区域（不与其他物体重叠）
    // - 设置目标点
    // - 切换到 PLAYING 状态
}

// 触摸事件处理
static void on_screen_touch(lv_event_t *e)
{
    lv_event_code_t code = lv_event_get_code(e);
    game_state_t state = game_manager_get_state();

    if (state == GAME_STATE_SET_TARGET && code == LV_EVENT_CLICKED) {
        lv_point_t point;
        lv_indev_get_point(lv_event_get_indev(e), &point);

        uint16_t map_x, map_y;
        if (renderer_screen_to_map_coord(&point, &map_x, &map_y)) {
            // 检查是否与物体重叠
            if (is_position_blocked(map_x, map_y)) {
                ESP_LOGI(TAG, "Target position (%u,%u) blocked by object", map_x, map_y);
                renderer_update_status("Position blocked! Tap another spot");
                return;
            }

            game_map_set_target(&g_ctx.game_map, map_x, map_y);
            renderer_update_target(map_x, map_y);
            renderer_update_status("Target set! Starting...");

            // 短暂延迟后开始游戏
            vTaskDelay(pdMS_TO_TICKS(1000));
            change_state(GAME_STATE_PLAYING);
        }
    }
    else if (state == GAME_STATE_RESTART && code == LV_EVENT_CLICKED) {
        change_state(GAME_STATE_INIT);
        game_manager_start();
    }
}
```

### STATE_PLAYING

```c
// 此状态由独立的 FreeRTOS 任务驱动

static void physics_task(void *arg)
{
    TickType_t last_wake = xTaskGetTickCount();
    const TickType_t period = pdMS_TO_TICKS(1000 / PHYSICS_UPDATE_HZ);

    while (1) {
        if (g_ctx.current_state != GAME_STATE_PLAYING) {
            vTaskDelay(pdMS_TO_TICKS(100));
            continue;
        }

        // 读取IMU倾斜数据（或使用模拟数据）
        float ax, ay;
#if IMU_ENABLED
        imu_get_tilt_accel(&ax, &ay);
#else
        // 模拟正弦变化的倾斜（调试用）
        static float t = 0;
        t += 1.0f / PHYSICS_UPDATE_HZ;
        ax = sinf(t * 0.5f) * 500.0f;
        ay = sinf(t * 0.3f) * 300.0f;
#endif

        physics_set_gravity(ax, ay);

        // 物理步进
        const marble_physics_t *marble = physics_step(1.0f / PHYSICS_UPDATE_HZ);

        // 更新渲染（在主线程/LVGL线程）
        bsp_display_lock(pdMS_TO_TICKS(10));
        marble_render_info_t info = {
            .map_x = marble->pos_x, .map_y = marble->pos_y, .radius = marble->radius
        };
        renderer_update_marble(&info);
        renderer_update_tilt_indicator(ax, ay);
        bsp_display_unlock();

        // 检查是否到达目标
        if (physics_check_goal_reached(g_ctx.game_map.target_x,
                                        g_ctx.game_map.target_y,
                                        GOAL_THRESHOLD)) {
            ESP_LOGI(TAG, "Goal reached!");
            change_state(GAME_STATE_GOAL_REACHED);
        }

        // 检查是否出界
        if (marble->pos_x < 0 || marble->pos_x > MAP_SIZE ||
            marble->pos_y < 0 || marble->pos_y > MAP_SIZE) {
            ESP_LOGW(TAG, "Marble out of bounds!");
            renderer_update_status("Marble fell off! Tap to restart");
            change_state(GAME_STATE_RESTART);
        }

        vTaskDelayUntil(&last_wake, period);
    }
}
```

## LVGL 触摸事件集成

```c
// 在渲染器初始化时注册触摸回调
void renderer_init(const renderer_config_t *cfg)
{
    // ... 创建UI元素 ...

    // 在地图容器上注册触摸事件
    lv_obj_add_event_cb(map_container, on_map_touch, LV_EVENT_CLICKED, NULL);

    // 或者在整个屏幕上注册（SET_TARGET/RESTART状态）
    lv_obj_t *scr = lv_disp_get_scr_act(NULL);
    lv_obj_add_event_cb(scr, on_screen_touch, LV_EVENT_CLICKED, NULL);
}

static void on_map_touch(lv_event_t *e)
{
    lv_event_code_t code = lv_event_get_code(e);
    if (code != LV_EVENT_CLICKED) return;

    lv_point_t point;
    lv_indev_get_point(lv_event_get_indev(e), &point);

    game_state_t state = game_manager_get_state();

    if (state == GAME_STATE_SET_TARGET) {
        uint16_t map_x, map_y;
        if (renderer_screen_to_map_coord(&point, &map_x, &map_y)) {
            // 处理目标设定
            handle_set_target(map_x, map_y);
        }
    } else if (state == GAME_STATE_RESTART || state == GAME_STATE_GOAL_REACHED) {
        // 重新开始
        handle_restart();
    }
}
```

## 状态切换函数

```c
static void change_state(game_state_t new_state)
{
    ESP_LOGI(TAG, "State: %d → %d", g_ctx.current_state, new_state);
    g_ctx.current_state = new_state;

    // 状态切换时的逻辑
    switch (new_state) {
        case GAME_STATE_CAPTURE:
            state_capture();
            break;
        case GAME_STATE_DETECT:
            state_detect();
            break;
        case GAME_STATE_MAP_SETUP:
            state_map_setup();
            break;
        case GAME_STATE_PLAYING:
            // 启动物理任务（如果尚未启动）
            if (g_ctx.physics_task_handle == NULL) {
                xTaskCreatePinnedToCore(physics_task, "physics",
                                        4096, NULL, 5,
                                        &g_ctx.physics_task_handle, 0);
            }
            break;
        case GAME_STATE_SET_TARGET:
        case GAME_STATE_GOAL_REACHED:
        case GAME_STATE_RESTART:
            // 无需特殊处理，等待用户交互
            break;
        default:
            break;
    }
}
```

## 主入口 (main.c)

```c
void app_main(void)
{
    ESP_LOGI(TAG, "=== AR Sandbox Starting ===");

    // 初始化 BSP（显示、触摸、LVGL）
    lv_display_t *disp = bsp_display_start();
    if (!disp) {
        ESP_LOGE(TAG, "Display init failed");
        return;
    }
    bsp_display_backlight_on();

    // 等待LVGL就绪
    vTaskDelay(pdMS_TO_TICKS(200));

    // 初始化游戏管理器
    bsp_display_lock(portMAX_DELAY);
    game_manager_init();
    bsp_display_unlock();

    // 开始游戏
    game_manager_start();

    ESP_LOGI(TAG, "=== Game started ===");
}
```

## 调试方法

1. **状态转换日志**：每次状态切换都输出日志
2. **触摸坐标日志**：记录触摸的屏幕坐标和转换后的地图坐标
3. **物理数据监控**：在状态栏显示实时弹珠位置和速度
4. **快捷调试键**：可以用特定触摸区域触发调试功能（如跳转到特定状态）
